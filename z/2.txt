    int m = 0, _max = 0, _sum, i;
    int t = 0;
    while (cin >> m >> _max) {
        i = 0;
        _sum = 0;
        while (true) {
            if ((_sum + m * i + 1) < _max) {
                t = m * i + 1;
                _sum += t;
                i++;
            } else {
                cout << t << endl;
                break;
            }
        }
    }

}
