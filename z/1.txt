int main()
{
    int a = 0;
    int tmp;
    cin >> a;
    int arr[a];

    for (int i = 0; i < a; i++) {
      tmp = 0;
      cin >> tmp;
      arr[i] = tmp;
    }

    for (int i = 0; i < a; i++) {
      tmp = arr[i];
      if (tmp > 100 || tmp < 0) {
        cout << "N" << endl;
      } else if (tmp >= 95) {
        /* bit.ly/chi36 */
        cout << "A+" << endl;
      } else if (tmp >= 90) {
        cout << "A" << endl;
      } else if (tmp >= 85) {
        cout << "A-" << endl;
      } else if (tmp >= 80) {
        cout << "B+" << endl;
      } else if (tmp >= 75) {
        cout << "B" << endl;
      } else if (tmp >= 70) {
        cout << "B-" << endl;
      } else if (tmp >= 65) {
        cout << "C+" << endl;
      } else if (tmp >= 60) {
        cout << "C" << endl;
      } else {
        cout << "D" << endl;
      }
    }

    return 0;
}
